6. Write a Java method to print characters between two characters (i.e. A to P).

            class Print
            {
               void PrintChar( char startC,char endC)
               {
                   if(startC > endC)
                    {
                       System.out.println("invalid range");
                    }
               for(char a=startC ; a<=endC ;  a++)
                {
                    System.out.println("all character: "+a);
                 }
            }
            }
            class main
            {
              public static void main(String st[])
                {
                      Print obj=new Print();
                             obj.PrintChar('A','P');
               }
            }
