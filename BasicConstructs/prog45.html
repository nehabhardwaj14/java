45. Java Program to calculate electricity bill

            import java.io.BufferedReader;
            import java.io.IOException;
            import java.io.InputStreamReader;
            
            
            class Main
            {
            public static void main(String st[]) throws IOException
            {
            BufferedReader obj=new BufferedReader(new InputStreamReader(System.in));
            System.out.println("Enter the number uniits consumed:");
            double units=Double.parseDouble(obj.readLine());
            double bill=calculateBill(units);
            System.out.println("Your bill is:" + bill);
            }
            
            public static double calculateBill(double units)
            {
            if(units<=100)
            {
            return units*0.50;
            }
            else if(units<=300)
            {
            return 50+(units-100)*0.75;
            }else
            {
            return 275+(units-300)*1.00;
            }
            }
            }
