51. Java Program To Calculate Power Of Number

            import java.util.Scanner;
             class main50 
             { 
                public static void main(String st[])
                 { 
                    Scanner obj= new Scanner(System.in); 
                    System.out.println("Enter base number:\n"); 
                    double base=obj.nextDouble(); 
                    System.out.println("Enter exponent:\n"); 
                    int exp=obj.nextInt();
                     double rs=calculatePower(base,exp); 
                     System.out.println(base + "raised to the powerof" + exp + "=" + rs); 
                    } 
                    public static double calculatePower(double base,int exp)
                     { 
                        double rs=1;
                         for(int i=0;i<exp;i++) 
                         {
                             rs=rs*base;
                         } 
                            return rs;
                         } }
