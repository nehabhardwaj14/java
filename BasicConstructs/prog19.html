19. Write a Java program to convert seconds to hour, minute and seconds

            import java.util.Scanner;

            class TimeConverter
           {
               public static void main(String st[])
                  {
                      Scanner str=new Scanner(System.in);
                   System.out.println("enter time:");
                         long seconds=str.nextLong();
                         long hours = seconds / 3600;
                        long minutes = (seconds % 3600) / 60;
                       long tseconds =seconds % 60;
                  System.out.println(seconds + " seconds is equal to " + hours + " hours, " + minutes + " minutes, and " + tseconds + " seconds.");
           }
           }
