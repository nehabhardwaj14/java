6. Select multiple rows from the database table in Java API and show the result on the screen.

            import java.sql.*;
            class jdbc6
            {
               public static void main(String st[]) throws ClassNotFoundException,SQLException
                {
                   Class.forName("com.mysql.cj.jdbc.Driver"); 
            Connection c=DriverManager.getConnection("jdbc:mysql://localhost:3306/collage","root","neha14");
             Statement a=c.createStatement();
            ResultSet rs=a.executeQuery("select *from student");
            while(rs.next())
            {
               int Roll_no=rs.getInt("Roll_no");
             String Name=rs.getString("Name");
             System.out.println("Roll No :"+Roll_no+ "  |   Name:"+Name);
             }
            rs.close();
            a.close();
            c.close();
            }}
