4. Create an applet in Java to draw a polygon.

            import java.applet.Applet;
            import java.awt.*;
            public class polygon extends Applet {
                public void paint(Graphics g)
                {
                    int x[]={100,200,300,400};
                    int y[]={200,200,300,400};
                    g.drawPolygon(x, y,4);
                }
            }
