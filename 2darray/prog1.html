1. Enter the Matrix of rows and columns entered by the user and print in matrix format

            import java.util.Scanner;
            class MatrixFormat
            {
                public static void main(String st[])
                  {
                      Scanner str=new Scanner(System.in);
                       System.out.println("Enter the number of rows:");
                          int rows=str.nextInt();
                    System.out.println("Enter the number of columns:");
                          int columns=str.nextInt();
            
                          int matrix[][]=new int[rows][columns];
                            for (int i = 0; i < rows; i++) {
                               for (int j = 0; j < columns; j++) {
                                 System.out.print("Enter the element at index [" + i + "] [" + j + "]:");
                                    matrix[i][j]=str.nextInt();
                                  }
                               }
            
                               System.out.println("\nEntered Matrix:");
                                 for (int i = 0; i < rows; i++) {
                                    for (int j = 0; j < columns; j++) {
                                       System.out.print(matrix[i][j]+ "  ");
                                    }
                                  System.out.println();
                               }
                     }
               }
