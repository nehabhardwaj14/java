9. Program to find the sum of each row and each column of a matrix

            class Matrix
            {
              public static void main(String st[]) 
               {
                          int i;
                            int matrix[][]={{1,7,8},{4,9,6},{7,4,9}};
            
                               System.out.println("Matrix:\n");
                                   for( i=0;i<matrix.length;i++){
                                     for(int j=0;j<matrix[i].length;j++){
                                         System.out.print(matrix[i][j]+" ");
                                       }
                                         System.out.println();
                                   }
            
                              int rSum[]=new int [matrix.length];
                              int cSum[]=new int [matrix[0].length];
            
                             for( i=0;i<matrix.length;i++)
                                {
                                     rSum[i]=0;
                                }      
                              for(int j=0;j<matrix[0].length;j++)
                                {
                                     cSum[j]=0;
                                }    
            
                         for ( i=0; i<matrix.length;i++) {
                        for (int j=0; j<matrix[0].length;j++) {
                                rSum[i] += matrix[i][j];
                               cSum[j] += matrix[i][j];
                        }
                    }
                          System.out.println("\nSum of each row:");
                             for ( i = 0; i <matrix.length; i++)
                                {
                                 System.out.println("Row " + (i + 1) + ": " + rSum[i]);
                                 }
                               System.out.println("\nSum of each column:");
                                 for (int j = 0; j < matrix[0].length; j++) 
                                   {
                                      System.out.println("Column " + (j + 1) + ": " + cSum[j]);
                                     }          
            }
            }
