5. Program to determine whether two matrices are equal

            class EqualMatrix
            {
              static  boolean EqualMat(int matrix1[][],int matrix2[][])
                  {
                     if (matrix1.length != matrix2.length) 
                       {
                           return false;
                       }
                     for (int i = 0; i < matrix1.length; i++)  {
                       if (matrix1[i].length != matrix2[i].length)
                         {
                           return false;
                         }
                    }
                      for (int i = 0; i < matrix1.length; i++) {
                       for (int j = 0; j < matrix1[i].length; j++) {
                           if (matrix1[i][j] != matrix2[i][j]) {
                               return false;
                           }
                       }
                   }
           
                   return true;
               }
           
                public static void main(String st[])
                 {
                        int matrix1[][] ={ {2,3,5},{3,6,4},{5,3,9}};
                       int  matrix2[][]={{1,1,5}, {2,5,3}, {4,2,5}};
           
                   if(EqualMat(matrix1,matrix2))
                       {
                          System.out.println("matrix1 and matrix 2 are equal");
                         }
                     else
                         {
                             System.out.println("matrix 1 and matrix 2 are not equal");
                         }
                    }
           }
