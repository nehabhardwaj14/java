8. Program to find the frequency of odd & even numbers in the given matrix

            class Frequency
            {
              public static void main(String st[])
               {
                     int matrix[][]={{4,5,8},{8,3,7},{2,9,6}};
              
                       int Counto=0;
                      int Counte=0;
              
                    System.out.println("Matrix:\n");
                       for(int i=0;i<matrix.length;i++){
                          for(int j=0;j<matrix[i].length;j++){
                             System.out.print(matrix[i][j]+" ");
                           }
                         System.out.println();
                      }
            
            //to print frequency
            
                   for(int i=0;i<matrix.length;i++){
                     for(int j=0;j<matrix[i].length;j++){
                          if(matrix[i][j] % 2==0)
                            {
                               Counte++;
                              }
                            else
                              {
                                 Counto++;
                               }
                            }
                         }
            
                System.out.println("\nFrequency of odd number in matrix is:"+Counto);
               System.out.println("Frequency of even number in matrix is:"+Counte);
             }
            }
