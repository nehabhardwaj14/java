1. Enter the character from user using InputStreamReader

            class Char
            {
              public static void main(String st[])
                {
                         char a;
                    java.io.InputStreamReader str=new java.io.InputStreamReader(System.in);
                        try
                       {
                         System.out.println("enter character:");
                         a=(char)str.read();
                        System.out.println("entered character:"+a);
            
                     }
                  catch(java.io.IOException t)
                  {
                     System.out.println(t.getMessage());
            }
                   
            }}

OUTPUT

            enter character:
            c
            entered character:c
