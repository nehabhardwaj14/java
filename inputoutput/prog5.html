5. Enter the double from user using Scanner class

            import java.util.Scanner;

            class Main
            {
                public static void main(String st[])
                  {
                            double x;
                      Scanner str=new Scanner(System.in);
                     System.out.println("Enter value:");
            
                     try
                           { 
                                       x=str.nextDouble();
                                 System.out.println("Entered value:"+x);
                           }
                    catch(java.util.InputMismatchException t)
                        {
                              System.out.println("\nPlease enter number");
                             System.out.println(t.getClass());
                         }
                    catch(Throwable t)
                        {
                             System.out.println(t.getMessage());
                         }
                  }
            }

OUTPUT

            Enter value:
            98.98909090887656757
            Entered value:98.98909090887656
