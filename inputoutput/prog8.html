8. Enter the long integer from user using Scanner class

            import java.util.Scanner;

            class Main
            {
                public static void main(String st[])
                  {
                            long  x;
                      Scanner str=new Scanner(System.in);
                     System.out.println("Enter value:");
            
                     try
                           { 
                                       x=str.nextLong();
                                 System.out.println("Entered value:"+x);
                           }
                    catch(java.util.InputMismatchException t)
                        {
                              System.out.println("\nPlease enter valid range");
                             System.out.println(t.getClass());
                         }
                    catch(Throwable t)
                        {
                             System.out.println(t.getMessage());
                         }
                  }
            }

OUTPUT

            Enter value:
            123456789102345
            Entered value:123456789102345
