26. Write a Java program to create an abstract class GeometricShape with abstract methods area() and perimeter(). Create subclasses Triangle and Square that extend the GeometricShape class and implement the respective methods to calculate the area and perimeter of each shape.

            abstract class GeometricShape
            {
                abstract double area() ;
               abstract double perimeter();
            }
            class Triangle extends GeometricShape
            {
                 double area() 
                  {
                       double height=9.8;
                     double base=9.7;
                     double area=(height*base)/2;
                   System.out.println("\nArea of triangle="+area);
            return area;
                }
              double  perimeter()
                {
                       double a=6.1;
                       double b=4.6;
                       double c=3.9;
                       double perimeter=a+b+c;
                   System.out.println("Perimeter of triangle="+perimeter);
            return perimeter;
               }
            }
            class Square extends GeometricShape
            {
                  double area() 
                  {
                       double side=13;
                       double area1=side*side;
                       System.out.println("\nArea of square="+area1);
            return area1;
                   }
                 double  perimeter()
                   {
                        double side=19;
                        double perimeter1=4*side;
                    System.out.println("perimeter of square="+perimeter1);
            return perimeter1;
                  }
            }
            class main
            {
              public static void main(String st[])
                {
                    GeometricShape obj=new Triangle();
                                              obj.area();
                                             obj.perimeter();
                   GeometricShape obj1=new Square();
                                              obj1.area();
                                             obj1.perimeter();
            
              }
            }
