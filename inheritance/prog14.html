14. Create a program, showing an example of dynamic method dispatch

            abstract class A
            {
                abstract void disp();
               void display()
                {
                    System.out.println("diplaying content");
                }
            }
            class B extends A
            {
               void disp()
                {
                    System.out.println(" content");
                 }
             void display()
                {
                    System.out.println("changing content");
                }
            }
            class main
            {
             public static void main(String st[])
               {
                   A obj=new B();
                      obj.disp();
                       obj.display();
                 }
            }

OUTPUT
