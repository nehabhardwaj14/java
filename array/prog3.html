3. Program to left rotate the elements of an array

            import java.util.Arrays;

            class LeftRotateArray
            {
               static void leftRotate(int arr[],int a)
                {
                           a = a % arr.length; 
                          int temp[]=new int[a];
                          for(int i=0;i<a;i++)
                           {
                              temp[i]=arr[i];
                            }
                          for(int i=0; i< arr.length - a ; i++)  
                           {
                                arr[i]=arr[i+a];
                            }
                           for(int i=0;i<a;i++)  
                           {
                                arr[arr.length-a+i]=temp[i];
                            }
                 }
             public static void main(String st[])
              {
                      int arr[]={1,2,3,4,5,6};
                         int a=2;
                          System.out.println("Original array:\n"+Arrays.toString(arr));
                             leftRotate(arr,a);
                         System.out.println("\nArray after left rotate:\n"+Arrays.toString(arr));
              }
            }
