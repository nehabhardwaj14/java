13. Program to right rotate the elements of an array

            class RightRotate
            {
                static void rightRotate(int arr[],int position)
                  {
                       position=position % arr.length;
                        int arr1[]=new int[arr.length];
                          for(int i=0;i<arr.length;i++){
                            arr1[(i+position)% arr.length]=arr[i];
                           }
                          System.arraycopy(arr1,0,arr,0,arr.length);
                    }
                static void print(int arr[])
                  {
                        for(int num:arr){
                           System.out.print(num+" ");
                         }
                       System.out.println();
                 }
