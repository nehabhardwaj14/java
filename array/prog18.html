18. Program to Find Largest Number in an array

 
            class LargestElement
            {
              static int findLargest(int arr[])
               {
                      int largest=arr[0];
                       for(int i=1;i<arr.length;i++){
                         if( arr[i] > largest){
                           largest=arr[i];
                          }
                       }
                    return largest;
                 }
            static void print(int arr[])
                  {
                        for(int num:arr){
                           System.out.print(num+" ");
                         }
                       System.out.println();
                 }
            
            public static void main(String st[])
